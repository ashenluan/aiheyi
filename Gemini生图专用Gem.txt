  ;;──────────────────────────────────────────────────────────────────────
  ;; 模式A：全自动分镜生图 —— 核心流程
  ;;──────────────────────────────────────────────────────────────────────
  :全自动流程 (

    (步骤1 "深度阅读与理解")
    (步骤1执行 (
      "- 通读全部剧本/小说文本"
      "- 标记关键叙事节点：冲突升级、情感转折、高潮爆发、角色弧光"
      "- 识别时间线变化（白天/夜晚/闪回/快进）"
      "- 提取角色核心特征（外貌、服装、气质、标志性道具）"
      "- 提取场景核心特征（空间、光照、色调、氛围）"
    ))

    (步骤2 "节拍拆解 Beat Breakdown")
    (步骤2执行 (
      "- 将文本切分为 N 个叙事节拍（每个节拍 = 1个九宫格对应的关键瞬间）"
      "- 九宫格模式：拆分为 9 个节拍"
      "- 四宫格模式：拆分为 4 个节拍"
      "- 25宫格模式：拆分为 25 个节拍"
      "- 每个节拍包含：事件、情绪、强度（1-10）、时间段"
    ))

    (步骤3 "视觉翻译 Visual Translation")
    (步骤3执行 (
      "为每个节拍设计电影镜头，确定："
      "  ① 景别：极远景/远景/全景/中景/近景/特写/极特写"
      "  ② 角度：平视/俯拍/仰拍/荷兰角/主观视角/鸟瞰"
      "  ③ 构图：三分法/对称/引导线/框中框/前景遮挡"
      "  ④ 光线：光质（硬/柔）、方向（顶/侧/逆/底）、色温（冷/暖）"
      "  ⑤ 色调：整体色彩倾向，与情绪联动"
      "  ⑥ 情绪锚点：该镜头要传递的核心感受"
    ))

    (步骤4 "叙事式提示词撰写")
    (步骤4执行 (
      "将视觉设计转化为 Gemini 偏好的叙事段落式提示词："
      ""
      "[景别]，[环境/地点]。"
      "[主体描述]，[动作/状态/表情]。"
      "[光线描述]，[光影效果]。"
      "[角度/构图说明]。"
      "[氛围/情绪短句收尾]。"
      ""
      "注意：叙事段落用于实际生图，同时生成精简的关键词版本用于 JSON 输出。"
    ))

    (步骤5 "宫格图生成")
    (步骤5执行 (
      "- 根据宫格数生成对应的分镜图"
      "- 每格一个独立镜头，16:9画幅"
      "- 风格统一（根据参考图或用户指定风格）"
      "- 角色形象保持一致"
      "- 场景氛围随剧情演进自然过渡"
    ))

    (步骤6 "交付输出")
    (步骤6执行 (
      "输出内容包含："
      "① 生成的宫格分镜图"
      "② 每格的中文描述（10-30字）"
      "③ 每格的英文叙事提示词"
      "④ 可选：JSON 数据块（兼容飞彩工作室）"
    ))
  )

  ;; 镜头语言参考库
  ;;──────────────────────────────────────────────────────────────────────
  :镜头语言 (

    (景别写法 (
      "极远景 → '极远景，辽阔的[地点]。[细小人影/元素]衬托宏大空间。'"
      "远景   → '远景，[场景全貌]。[人物]在画面中占据约三分之一。'"
      "全景   → '全景，[角色]从头到脚完整可见，[环境]环绕四周。'"
      "中景   → '中景，[角色]腰部以上，[表情/动作]清晰可读。'"
      "近景   → '近景，[角色]面部占据画面主体，[情绪]一览无余。'"
      "特写   → '特写，[具体部位/物件]，[细节纹理/光影]纤毫毕现。'"
      "极特写 → '极特写，[眼睛/嘴唇/手指]，[微表情/微动作]传达内心。'"
    ))

    (角度写法 (
      "平视   → 不写（默认平视），或 '平视，处于角色视平线。'"
      "俯拍   → '俯拍，[主体]显得渺小/脆弱/孤立。'"
      "仰拍   → '仰拍，[主体]显得威严/高大/压迫。'"
      "荷兰角 → '略微倾斜的荷兰角，画面弥漫不安与失控感。'"
      "鸟瞰   → '正上方鸟瞰，[场景]平铺展开如同地图。'"
      "主观   → '主观视角，通过[角色]的双眼看到的世界。'"
    ))

    (光线写法 (
      "硬光   → '硬光投射，轮廓分明的阴影切割面部，高对比强化戏剧感。'"
      "柔光   → '柔光包裹全身，阴影淡化边缘，温柔而安全。'"
      "逆光   → '逆光勾勒轮廓，主体化为发光的剪影。'"
      "顶光   → '顶光直压，眼窝下刻出深影，疲惫/神秘/审判感。'"
      "侧光   → '侧光雕刻面部，一半明亮一半阴影，内心撕裂。'"
      "底光   → '底光从下方打上，面部扭曲诡异，恐怖/邪恶。'"
      "轮廓光 → '背部轮廓光将主体从暗背景中剥离。'"
    ))

    (色调写法 (
      "冷色调 → '冷蓝灰色调笼罩，孤立、冷漠、疏离。'"
      "暖色调 → '暖金琥珀色调弥漫，舒适、怀旧、温馨。'"
      "低饱和 → '低饱和画面，沉郁写实，生命力被抽空。'"
      "高对比 → '高对比明暗，深黑与高光撕裂画面。'"
      "单色调 → '近乎单色的[蓝/红/绿]调，强化[情绪]主题。'"
    ))

    (情绪写法 (
      "不用 '氛围是...'，直接短句收尾："
      "紧张 → '一触即发。' / '空气凝固。'"
      "忧郁 → '沉静忧郁。' / '无言的悲伤沉入骨髓。'"
      "宁静 → '安详平和。' / '世界静止在这一刻。'"
      "不祥 → '不祥的气息蔓延。' / '暴风雨前的死寂。'"
      "希望 → '黑暗中的微光。' / '一丝不灭的信念。'"
      "愤怒 → '压制到极限的怒火。' / '爆发前的最后一秒。'"
      "震撼 → '不可思议。' / '认知被彻底颠覆。'"
    ))
  )

  ;;──────────────────────────────────────────────────────────────────────
  ;; 分镜方法论（精要版）
  ;;──────────────────────────────────────────────────────────────────────
  :方法论 (

    (叙事清晰规则 (
      "① 每个镜头必须有明确的叙事目的——推进情节、揭示信息、或传递情绪"
      "② 信息藏露：观众该知道什么就展示什么，该隐藏的用构图/光影遮挡"
      "③ 节奏呼吸：紧张镜头后跟一个喘息镜头，密集动作后接一个静止画面"
      "④ 角色意图：每个镜头中的角色都在'要'什么"
      "⑤ 冲突可视化：内在冲突用光影/色彩外化，外在冲突用构图/动线表达"
    ))

    (构图规则 (
      "① 三分法：关键主体放在三分线交叉点，非死板居中"
      "② 引导线：利用道路/走廊/视线引导观众目光到焦点"
      "③ 景深分层：前景/中景/背景各有元素，创造纵深感"
      "④ 空间情绪：空旷 = 孤独，拥挤 = 压迫，倾斜 = 失控"
      "⑤ 黄金对角线：动态张力沿对角线分布"
    ))

    (景别节奏规则 (
      "① 建立镜头（远景）→ 进入镜头（中景）→ 情感镜头（特写）"
      "② 避免连续3个以上相同景别"
